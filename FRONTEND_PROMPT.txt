Analytics:

- GET /api/stock-alerts/reorder-report
- PUT /api/stock-alerts/reorder/:id (body: { quantity })

Top Sellers:
- GET /api/top-sellers/products

Reviews:
- GET /api/reviews (query: search, productId, rating, sortBy, sortOrder, page, limit)

Notifications:
- GET /api/notifications (query: type, isRead, sortBy, sortOrder, page, limit)
- GET /api/notifications/unread-count

Export:
